% ------ 1.1 Raw Data ------
\begin{frame}{Raw Participant-Level Data}
Two matrices share matched participant rows ($N_1 \ldots N_n$):

\medskip
\hspace{9em}\textbf{Structural MRI} ($n\times m$) \hspace{6em} \textbf{Genotype} ($n\times l$)

\[
\mathbf{X}_{\text{MRI}} =
\begin{pmatrix}
\text{MRI}_{1,1} & \cdots & \text{MRI}_{1,m}\\
\vdots & \ddots & \vdots\\
\text{MRI}_{n,1} & \cdots & \text{MRI}_{n,m}
\end{pmatrix}
\qquad
\mathbf{X}_{G} =
\begin{pmatrix}
G_{1,1} & \cdots & G_{1,l}\\
\vdots & \ddots & \vdots\\
G_{n,1} & \cdots & G_{n,l}
\end{pmatrix}
\]

\begin{itemize}
    \item $n$ = number of participants (European ancestry cohort)
    \item $m$ = MRI phenotypes (volume, area, thickness, intensity, contrast per brain region)
    \item $l$ = genotyped SNP variants
\end{itemize}
\end{frame}

% ------ 1.2 GWAS Transformation ------
\begin{frame}{GWAS Transformation}
For every SNP $i$ and MRI phenotype $j$, a GWAS collapses the $n$ participants into a single z-score:

\[
\text{GWAS}\!\bigl(\mathbf{X}_{G_{*,i}},\;\mathbf{X}_{\text{MRI}_{*,j}}\bigr)
\;\longrightarrow\; z_{i,j}
\]
Under a linear model for quantitative traits:
\[
y = G_{\text{MRI}_{*,j}}\,\beta^{G}_{i,j} + X\beta^{X} + e
\]

The z-score is
$\displaystyle z_{i,j} = \frac{\beta^{G}_{i,j}}{\mathrm{s.e.}(\beta^{G}_{i,j})}$.

\medskip
\begin{itemize}
    \item $\beta^{G}_{i,j}$: effect size of SNP $i$ on phenotype $j$
    \item $X,\beta^{X}$: covariates (age, sex, PCs\ldots)
    \item p-value: $\chi^{2} = \beta^{2}\cdot N$, then inverted
\end{itemize}
\end{frame}

% ------ 1.3 Final Dataset ------
\begin{frame}{Final Tabular ML Dataset (per illness)}
Rows = SNPs, features = MRI z-scores, target = illness z-score.

\[
\left[\;
\begin{array}{cccc|c}
z_{1,1} & z_{1,2} & \cdots & z_{1,m} & z_{\text{ill},1}\\
z_{2,1} & z_{2,2} & \cdots & z_{2,m} & z_{\text{ill},2}\\
\vdots  & \vdots  & \ddots & \vdots  & \vdots\\
z_{l,1} & z_{l,2} & \cdots & z_{l,m} & z_{\text{ill},l}
\end{array}
\;\right]
\begin{array}{l}
\leftarrow G_1\\
\leftarrow G_2\\
\vdots\\
\leftarrow G_l
\end{array}
\]

\begin{itemize}
    \item After clumping: $l \approx 3{,}500$ SNPs, $m \approx 1{,}000$ MRI features
    \item SCZ has highest predictive power (largest GWAS sample size)
    \item Negative median z-score is expected: allele ordering in GWAS is arbitrary
\end{itemize}
\end{frame}


% ------ 1.4 Clumping & LD ------
\begin{frame}{P-Value Clumping \& Linkage Disequilibrium}
\begin{block}{Linkage Disequilibrium (LD)}
\begin{itemize}\setlength\itemsep{3pt}
    \item Nearby nucleotides are correlated because recombination is less likely at short distances.
    \item $\mathrm{E}[\hat\beta_i] = \beta_i + \sum_{j\neq i} r_{ij}\beta_j$ (LD bias within a block).
    \item LD blocks are estimated in \textsc{PLINK} 
    \item correlation of chosen SNP with other SNPsof LD block is removed.
\end{itemize}
\end{block}

\begin{block}{Clumping}
\begin{itemize}\setlength\itemsep{3pt}
    \item Pick the lead SNP per LD block across all illnesses
    \item P-value threshold (e.g.\ $p < 10^{-3}$) controls how many SNPs survive 
    \newline $\Rightarrow$ dataset size vs.\ signal quality trade-off.
    \item Same SNP set selected across illnesses \newline $\Rightarrow$ cross-illness comparison is valid.
\end{itemize}
\end{block}
\end{frame}

\section{Goal \& Task Formulation}
% ============================================================

\begin{frame}{ML Objective}
Learn a mapping:
\[
f(\mathbf{z}_i) \approx y_i,
\qquad
\mathbf{z}_i = [z_{i,1},\dots,z_{i,m}] \in \mathbb{R}^{m},
\quad
y_i \in \mathbb{R}
\]

\textbf{Two formulations:}
\begin{enumerate}
    \item \textbf{Regression}: predict the continuous illness z-score directly.\\
          $f\colon\mathbb{R}^{m}\to\mathbb{R}$
    \item \textbf{Binary classification}: threshold the z-score $\Rightarrow$ risk / no-risk label.\\
          Potentially more feasible given label distribution.
\end{enumerate}

\medskip
Both are instances of \emph{tabular machine learning}, which remains notoriously hard for unstructured feature sets \cite{kadraWelltunedSimpleNets, holzmullerBetterDefaultStrong2024}.
\end{frame}

\begin{frame}{TabPFN -- Bayesian View and PFN Objective}
\scriptsize
\textbf{Posterior predictive distribution (PPD):}
for a test sample $x_{\text{test}}$, the model predicts
$p(\cdot \mid x_{\text{test}}, D_{\text{train}})$ conditioned on
$D_{\text{train}} := \{(x_1,y_1),\ldots,(x_n,y_n)\}$.

\medskip
The Bayesian PPD integrates over hypotheses $\phi \in \Phi$:
\[
p(y \mid x, D) \propto \int_{\Phi} p(y \mid x, \phi)\,p(D \mid \phi)\,p(\phi)\,d\phi,
\qquad \phi \in \Phi
\]

TabPFN approximates this posterior predictive with a transformer $q_\theta$ trained via
\[
\mathcal{L}_{\text{PFN}}=
\mathbb{E}_{(\{(x_{\text{test}},y_{\text{test}})\}\cup D_{\text{train}})\sim p(D)}
\left[-\log q_\theta\!\big(y_{\text{test}}\mid x_{\text{test}},D_{\text{train}}\big)\right].
\]

Data is sampled from the prior predictive process:
\[
p(D)=\mathbb{E}_{\phi\sim p(\phi)}\!\big[p(D\mid\phi)\big],
\qquad
\phi\sim p(\phi),\; D\sim p(D\mid\phi).
\]

At inference, we input $\langle D_{\text{train}},x_{\text{test}}\rangle$ and obtain
\[
q_\theta\big(y\mid x_{\text{test}},D_{\text{train}}\big)
\]
as a single forward-pass approximation of the PPD. Further information can be found in \cite{hollmannTabPFNTransformerThat2023}.

\end{frame}

\section{Next Steps}
% ============================================================

\begin{frame}{Next Steps}
\begin{enumerate}\setlength\itemsep{6pt}
    \item \textbf{Data distribution}\\
        Improve label distribution, represent whole range of z-scores

    \item \textbf{Regularisation Cocktail MLP}\\
          Systematically search implicit, structural, data augmentation, and ensemble methods \cite{kadraWelltunedSimpleNets}. Will take a while to train, no guarantee of improvement.

    \item \textbf{TabPFNv2 optimization}\\
          Hyperoptimization, fine-tuning and continued pre-training for TabPFNv2. Inspired by \cite{hollmannAccuratePredictionsSmall2025}. Promising avenue to boost performance.

    \item \textbf{TabPFNv2 Explainability}\\
          Use SHAP values to identify which MRI features drive predictions; compare with known biology. Create causal graphs of feature importance.

    \item \textbf{Graph Neural Network}\\
          Treat brain regions as nodes with MRI features; edge definition TBD (spatial adjacency, functional connectivity, or learned).


\end{enumerate}
\end{frame}
